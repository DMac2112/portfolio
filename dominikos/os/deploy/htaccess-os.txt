# Copy this file to /os/.htaccess on Apache hosts (DOMINIKOS-PLAN §12.4).
RewriteEngine On
RewriteBase /os/
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule . /os/index.html [L]

<IfModule mod_headers.c>
  <FilesMatch "index\.html$">
    Header set Cache-Control "no-cache"
    Header set Content-Security-Policy "frame-src 'self' https://dominikmachowiak.com; frame-ancestors 'self'"
  </FilesMatch>
  <FilesMatch "\.(js|css|woff2|svg|png|webp)$">
    Header set Cache-Control "public, max-age=31536000, immutable"
  </FilesMatch>
</IfModule>
